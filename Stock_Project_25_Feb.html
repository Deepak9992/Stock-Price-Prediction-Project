def getsummary(year, col_name):
    print('Show Pivot for Year {0} and Columns {1}'.format(year, col_name))
    dfa=df[df['years']==year]
    dt=pd.pivot_table(dfa, index='Months', columns='Days',
                  values=col_name, 
                  aggfunc='mean', fill_value=0)
    return dt
